cmake_minimum_required(VERSION 3.10)
project(AsteroidsGame)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Force Visual Studio toolchain
set(CMAKE_GENERATOR_TOOLSET "v143" CACHE STRING "Visual Studio toolset" FORCE)

# Windows-specific settings
if(WIN32)
    # Set output directories
    set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
    set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    
    # Windows-specific compiler flags
    add_compile_options(/W4 /EHsc)
    
    # Set Windows subsystem
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:WINDOWS")
    
    # Find GLFW (try multiple locations)
    set(GLFW_DIRS
        "${CMAKE_SOURCE_DIR}/third_party/glfw"
        "C:/vcpkg/installed/x64-windows"
        "C:/Program Files/GLFW"
    )
    
    find_path(GLFW_INCLUDE_DIR NAMES GLFW/glfw3.h PATHS ${GLFW_DIRS} PATH_SUFFIXES include)
    find_library(GLFW_LIBRARY NAMES glfw3 glfw3dll PATHS ${GLFW_DIRS} PATH_SUFFIXES lib)
    
    if(NOT GLFW_INCLUDE_DIR OR NOT GLFW_LIBRARY)
        message(FATAL_ERROR "GLFW not found. Please install GLFW or set GLFW_DIR")
    endif()
endif()

# Find required packages
find_package(OpenGL REQUIRED)

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/include)
include_directories(${CMAKE_SOURCE_DIR}/third_party/bgfx/include)
include_directories(${CMAKE_SOURCE_DIR}/third_party/miniaudio)
include_directories(${GLFW_INCLUDE_DIR})

# Source files
set(SOURCES
    src/main.cpp
    src/game.cpp
    src/renderer.cpp
    src/audio.cpp
    src/entity.cpp
    src/player.cpp
    src/asteroid.cpp
    src/bullet.cpp
    src/particle_system.cpp
    src/shader_utils.cpp
)

# Create executable
add_executable(${PROJECT_NAME} ${SOURCES})

# Link libraries
if(WIN32)
    target_link_libraries(${PROJECT_NAME} 
        ${GLFW_LIBRARY}
        OpenGL::GL
        kernel32
        user32
        gdi32
        winspool
        shell32
        ole32
        oleaut32
        uuid
        comdlg32
        advapi32
    )
    
    # Copy GLFW DLL to output directory if using dynamic linking
    if(GLFW_LIBRARY MATCHES ".*glfw3dll.*")
        get_filename_component(GLFW_DLL_DIR ${GLFW_LIBRARY} DIRECTORY)
        file(GLOB GLFW_DLL "${GLFW_DLL_DIR}/glfw3.dll")
        if(GLFW_DLL)
            add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E copy_if_different
                ${GLFW_DLL} $<TARGET_FILE_DIR:${PROJECT_NAME}>)
        endif()
    endif()
else()
    target_link_libraries(${PROJECT_NAME} 
        glfw 
        OpenGL::GL
        m
        dl
        pthread
    )
endif()

# Compiler flags
target_compile_options(${PROJECT_NAME} PRIVATE
    $<$<CXX_COMPILER_ID:MSVC>:/W4 /EHsc>
    $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -O2>
)

# Set debugger working directory
set_target_properties(${PROJECT_NAME} PROPERTIES
    VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}"
)

# Windows-specific properties
if(WIN32)
    set_target_properties(${PROJECT_NAME} PROPERTIES
        WIN32_EXECUTABLE TRUE
        VS_DEBUGGER_ENVIRONMENT "PATH=${CMAKE_SOURCE_DIR}/bin;$ENV{PATH}"
    )
endif()
