cmake_minimum_required(VERSION 3.10)
project(AsteroidsGame)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Windows-specific settings
if(WIN32)
    # Set output directories
    set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
    set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    
    # Windows-specific compiler flags
    add_compile_options(/W4)
    
    # Find MSYS2/MinGW packages
    set(CMAKE_PREFIX_PATH "C:/msys64/mingw64")
endif()

# Find required packages
find_package(PkgConfig REQUIRED)
find_package(OpenGL REQUIRED)

# Find GLFW
find_package(glfw3 REQUIRED)

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/include)
include_directories(${CMAKE_SOURCE_DIR}/third_party/bgfx/include)
include_directories(${CMAKE_SOURCE_DIR}/third_party/miniaudio)

# Source files
set(SOURCES
    src/main.cpp
    src/game.cpp
    src/renderer.cpp
    src/audio.cpp
    src/entity.cpp
    src/player.cpp
    src/asteroid.cpp
    src/bullet.cpp
    src/particle_system.cpp
    src/shader_utils.cpp
)

# Create executable
add_executable(${PROJECT_NAME} ${SOURCES})

# Link libraries
if(WIN32)
    target_link_libraries(${PROJECT_NAME} 
        glfw 
        OpenGL::GL
        kernel32
        user32
        gdi32
        winspool
        shell32
        ole32
        oleaut32
        uuid
        comdlg32
        advapi32
    )
else()
    target_link_libraries(${PROJECT_NAME} 
        glfw 
        OpenGL::GL
        m
        dl
        pthread
    )
endif()

# Compiler flags
target_compile_options(${PROJECT_NAME} PRIVATE
    $<$<CXX_COMPILER_ID:MSVC>:/W4>
    $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -O2>
)

# Set debugger working directory
set_target_properties(${PROJECT_NAME} PROPERTIES
    VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}"
)
